123  std::shared_ptr<T> search(T &start) {
124  int giveUpCount = MAX_SEARCH_NODES;
125 
126  while (!open.empty() && --giveUpCount >= 0) {
127  std::shared_ptr<T> curOpen{open[0]};
128  std::pop_heap(open.begin(), open.end(),
129  [](std::shared_ptr<T> &n1, std::shared_ptr<T> &n2) {
130  return *n1 > *n2;
131  });
132  open.pop_back();
133 
134  for (T &c : curOpen->get_available_nodes(curOpen)) {
135  std::shared_ptr<T> child{std::make_shared<T>(c)};
136 
137  auto result = closed.emplace(child);
138  if (result.second) {
139  open.emplace_back(*result.first);
140  std::push_heap(
141  open.begin(), open.end(),
142  [](std::shared_ptr<T> &n1, std::shared_ptr<T> &n2) {
143  return *n1 > *n2;
144  });
145  }
146  if (*child == start) {
147  return *closed.find(child);
148  }
149  }
150  }
151  return nullptr;
152  }
